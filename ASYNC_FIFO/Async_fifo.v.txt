module async_fifo #(
    parameter WIDTH = 8,
    parameter DEPTH = 16,
    parameter ADDR_SIZE = 4
)(
    // Write side
    input  logic             wr_clk,
    input  logic             wr_rst_n,
    input  logic             wr_en,
    input  logic [WIDTH-1:0] wr_data,

    // Read side
    input  logic             rd_clk,
    input  logic             rd_rst_n,
    input  logic             rd_en,
    output logic [WIDTH-1:0] rd_data,

    output logic full,
    output logic empty
);

///////////////////////////////////////////////////////////
// MEMORY ARRAY
///////////////////////////////////////////////////////////

logic [WIDTH-1:0] mem [0:DEPTH-1];

///////////////////////////////////////////////////////////
// BINARY AND GRAY POINTERS
///////////////////////////////////////////////////////////

logic [ADDR_SIZE:0] wr_ptr_bin, wr_ptr_gray;
logic [ADDR_SIZE:0] rd_ptr_bin, rd_ptr_gray;

///////////////////////////////////////////////////////////
// SYNCHRONIZED POINTERS
///////////////////////////////////////////////////////////

logic [ADDR_SIZE:0] rd_ptr_gray_sync1, rd_ptr_gray_sync2;
logic [ADDR_SIZE:0] wr_ptr_gray_sync1, wr_ptr_gray_sync2;

///////////////////////////////////////////////////////////
// BINARY TO GRAY FUNCTION
///////////////////////////////////////////////////////////

function automatic [ADDR_SIZE:0] bin2gray(
    input [ADDR_SIZE:0] bin
);
begin
    bin2gray = (bin >> 1) ^ bin;
end
endfunction

///////////////////////////////////////////////////////////
// WRITE DOMAIN
///////////////////////////////////////////////////////////

always_ff @(posedge wr_clk or negedge wr_rst_n) begin
    if (!wr_rst_n) begin
        wr_ptr_bin  <= 0;
        wr_ptr_gray <= 0;
    end
    else if (wr_en && !full) begin

        // Write data into memory
        mem[wr_ptr_bin[ADDR_SIZE-1:0]] <= wr_data;

        // Increment binary pointer
        wr_ptr_bin <= wr_ptr_bin + 1;

        // Convert to Gray code
        wr_ptr_gray <= bin2gray(wr_ptr_bin + 1);
    end
end

///////////////////////////////////////////////////////////
// READ DOMAIN
///////////////////////////////////////////////////////////

always_ff @(posedge rd_clk or negedge rd_rst_n) begin
    if (!rd_rst_n) begin
        rd_ptr_bin  <= 0;
        rd_ptr_gray <= 0;
        rd_data     <= 0;
    end
    else if (rd_en && !empty) begin

        // Read data
        rd_data <= mem[rd_ptr_bin[ADDR_SIZE-1:0]];

        // Increment pointer
        rd_ptr_bin <= rd_ptr_bin + 1;

        // Gray conversion
        rd_ptr_gray <= bin2gray(rd_ptr_bin + 1);
    end
end

///////////////////////////////////////////////////////////
// SYNCHRONIZE READ POINTER INTO WRITE CLOCK DOMAIN
///////////////////////////////////////////////////////////

always_ff @(posedge wr_clk or negedge wr_rst_n) begin
    if (!wr_rst_n) begin
        rd_ptr_gray_sync1 <= 0;
        rd_ptr_gray_sync2 <= 0;
    end
    else begin
        rd_ptr_gray_sync1 <= rd_ptr_gray;
        rd_ptr_gray_sync2 <= rd_ptr_gray_sync1;
    end
end

///////////////////////////////////////////////////////////
// SYNCHRONIZE WRITE POINTER INTO READ CLOCK DOMAIN
///////////////////////////////////////////////////////////

always_ff @(posedge rd_clk or negedge rd_rst_n) begin
    if (!rd_rst_n) begin
        wr_ptr_gray_sync1 <= 0;
        wr_ptr_gray_sync2 <= 0;
    end
    else begin
        wr_ptr_gray_sync1 <= wr_ptr_gray;
        wr_ptr_gray_sync2 <= wr_ptr_gray_sync1;
    end
end

///////////////////////////////////////////////////////////
// EMPTY LOGIC
///////////////////////////////////////////////////////////

assign empty = (rd_ptr_gray == wr_ptr_gray_sync2);

///////////////////////////////////////////////////////////
// FULL LOGIC
///////////////////////////////////////////////////////////

assign full =
    (wr_ptr_gray ==
    {~rd_ptr_gray_sync2[ADDR_SIZE:ADDR_SIZE-1],
      rd_ptr_gray_sync2[ADDR_SIZE-2:0]});

Endmodule
